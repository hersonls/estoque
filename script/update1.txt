alter table produtos drop lote;

CREATE VIEW VPRODUTOS
AS
SELECT
  PRODUTOS.CODIGO,
  PRODUTOS.NOME,
  PRODUTOS.CATEGORIA,
  CATEGORIA.NOME AS NOMECATEGORIA,
  PRODUTOS.CODBARRAS,
  PRODUTOS.MARCA,
  MARCA.NOME AS NOMEMARCA,
  PRODUTOS.PCUSTO,
  PRODUTOS.PVENDA,
  PRODUTOS.RUA,
  PRODUTOS.PRATELEIRA,
  PRODUTOS.SECAO,
  PRODUTOS.ESTOQUE,
  PRODUTOS.ESTOQUE_MIN
FROM
  PRODUTOS
  INNER JOIN MARCA ON (MARCA.CODIGO = PRODUTOS.MARCA)
  INNER JOIN CATEGORIA ON (CATEGORIA.CODIGO = PRODUTOS.CATEGORIA);
